<script>

const PASS_THRESH = 350, BORDERLINE = 420;

const inputs = {

  bv:   { active: true,  eq: x => 10.622*x - 348.37, r2: 0.4308 },

  dxit: { active: true,  eq: x => 2.3419*x + 374.49, r2: 0.4957 },

  abr:  { active: true,  eq: x => 9.5445*x - 259.67, r2: 0.5447 },

  rpi:  { active: false, eq: x => 6.0182*x + 104.04, r2: 0.2477 },

  rpb:  { active: false, eq: x => 3.1496*x + 284.38, r2: 0.0762 },

  is:   { active: false, eq: x => 1.3780*x + 383.04, r2: 0.3350 },

};

function toggleInput(key) {

  inputs[key].active = !inputs[key].active;

  document.getElementById('tog-' + key).classList.toggle('on', inputs[key].active);

  document.getElementById('card-' + key).classList.toggle('disabled', !inputs[key].active);

  document.getElementById('sl-' + key).disabled = !inputs[key].active;

  update();

}

function getVal(k) { return parseFloat(document.getElementById('sl-' + k).value); }

function update() {

  ['bv','dxit','abr','rpi','rpb','is'].forEach(k => {

    const v = getVal(k);

    document.getElementById('disp-' + k).textContent = (k==='dxit'||k==='is') ? v : v+'%';

  });

  const active = Object.keys(inputs).filter(k => inputs[k].active);

  document.getElementById('n-active').textContent = active.length;

  if (!active.length) { document.getElementById('pred-score').textContent='—'; document.getElementById('pred-range').textContent=''; document.getElementById('pass-pill').innerHTML=''; document.getElementById('gauge-fill').style.width='0%'; document.getElementById('avg-r2').textContent='—'; return; }

  let wSum=0, pSum=0;

  active.forEach(k => { const p=inputs[k].eq(getVal(k)); pSum+=p*inputs[k].r2; wSum+=inputs[k].r2; });

  const pred=Math.round(pSum/wSum), avgR2=(wSum/active.length).toFixed(2);

  const sd=Math.round(94*Math.sqrt(1-wSum/active.length));

  document.getElementById('pred-score').textContent=pred;

  document.getElementById('pred-range').textContent=Math.round(pred-sd)+' – '+Math.round(pred+sd)+' (±1 SD)';

  document.getElementById('avg-r2').textContent=avgR2;

  const pct=Math.min(100,Math.max(0,(pred-200)/600*100));

  const fill=document.getElementById('gauge-fill');

  fill.style.width=pct.toFixed(1)+'%';

  fill.style.background=pred>=BORDERLINE?'#1D9E75':pred>=PASS_THRESH?'#BA7517':'#E24B4A';

  const pill=document.getElementById('pass-pill');

  pill.innerHTML=pred>=BORDERLINE?'<span class="pass-pill pass">Likely pass</span>':pred>=PASS_THRESH?'<span class="pass-pill borderline">Borderline</span>':'<span class="pass-pill fail">At risk</span>';

}

update();

</script>
